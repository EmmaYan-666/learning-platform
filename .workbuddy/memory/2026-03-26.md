**代码结构**：
```javascript
async function speakNatural(text, lang) {
  try {
    if (azureEnabled && window.SpeechSDK) {
      await speakWithAzure(text, lang);
    } else {
      speak(text, lang); // 回退
    }
  } catch (error) {
    speak(text, lang); // 失败回退
  }
}
```
